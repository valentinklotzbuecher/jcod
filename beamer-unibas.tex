% Clean minimal beamer theme - Deep Space Purple
% Matches the HTML theme-dark.scss

% Colors (from theme-dark.scss)
\definecolor{deepspace}{HTML}{0f0a1a}      % $body-bg - deep space purple-black
\definecolor{lavenderwhite}{HTML}{e8e4f0}  % $body-color - soft lavender white
\definecolor{vibrantpurple}{HTML}{9D6EE8}  % $primary - vibrant purple accent
\definecolor{surface}{HTML}{1a1225}        % $surface - card backgrounds
\definecolor{purplehover}{HTML}{B48EF0}    % $primary-hover

% Set colors
\setbeamercolor{background canvas}{bg=deepspace}
\setbeamercolor{normal text}{fg=lavenderwhite}
\setbeamercolor{frametitle}{fg=white}
\setbeamercolor{title}{fg=white}
\setbeamercolor{subtitle}{fg=vibrantpurple}
\setbeamercolor{author}{fg=lavenderwhite}
\setbeamercolor{date}{fg=lavenderwhite}
\setbeamercolor{item}{fg=vibrantpurple}
\setbeamercolor{block title}{fg=vibrantpurple}
\setbeamercolor{structure}{fg=vibrantpurple}
\setbeamercolor{alerted text}{fg=purplehover}

% Code block colors
\setbeamercolor{block body}{bg=surface}

% Remove navigation symbols
\setbeamertemplate{navigation symbols}{}

% Simple frame title with purple accent line
\setbeamertemplate{frametitle}{
  \vspace{0.5em}
  \insertframetitle
  \par
  \vspace{-0.3em}
  \textcolor{vibrantpurple}{\rule{2cm}{1.5pt}}
}

% Clean footline with page numbers
\setbeamertemplate{footline}{
  \hfill
  \textcolor{lavenderwhite!50}{\footnotesize\insertframenumber/\inserttotalframenumber}
  \hspace{1em}
  \vspace{0.5em}
}

% Itemize styling - purple bullets
\setbeamertemplate{itemize item}{\textcolor{vibrantpurple}{\textbullet}}
\setbeamertemplate{itemize subitem}{\textcolor{vibrantpurple}{\textendash}}
\setbeamertemplate{itemize subsubitem}{\textcolor{vibrantpurple}{\guillemotright}}

% Enumerate styling
\setbeamertemplate{enumerate item}{\textcolor{vibrantpurple}{\insertenumlabel.}}

% Table of contents styling
\setbeamercolor{section in toc}{fg=lavenderwhite}
\setbeamercolor{subsection in toc}{fg=lavenderwhite!70}

% Hyperlink colors
\hypersetup{
  colorlinks=true,
  linkcolor=vibrantpurple,
  urlcolor=purplehover,
  citecolor=vibrantpurple
}
